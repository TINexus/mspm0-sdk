964 __STATIC_INLINE uint32_t DL_FlashCTL_getFlashSectorNumberInBank(
965  FLASHCTL_Regs *flashctl, uint32_t addr)
966 {
967  uint8_t numBanks = DL_FactoryRegion_getNumBanks();
968  uint32_t mainFlashSize = DL_FactoryRegion_getMAINFlashSize();
969  uint32_t sector, bankSectors, sectorInBank;
970 
971  /* Current sector over the whole memory map */
972  sector = DL_FlashCTL_getFlashSectorNumber(flashctl, addr);
973  if (numBanks > 1) {
974  /* Number of sectors per bank, considered max sector
975  * count for a bank-adjusted sector. Assume banks are
976  * evenly distributed */
977  bankSectors = (mainFlashSize / numBanks);
978  /* We will not assume that the maximum number of bank sectors is a
979  * multiple of 2. Which does mean incurring a modulo operation. */
980  sectorInBank = sector % bankSectors;
981  } else {
982  sectorInBank = sector;
983  }
984 
985  return sectorInBank;
986 }
