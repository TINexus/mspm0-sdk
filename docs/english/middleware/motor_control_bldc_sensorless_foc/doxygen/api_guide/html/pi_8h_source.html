134 __STATIC_INLINE void PI_run(PI_Instance *handle)
135 {
136  /* proportional term */
137  handle->up = _IQmpy(handle->Kp, (handle->Ref - handle->Fbk));
138 
139  /* integral term */
140  if(handle->Out == handle->v1)
141  {
142  handle->ui = _IQrepeat(handle->Ki, handle->up)+ handle->i1;
143  }
144  else
145  {
146  handle->ui = handle->i1;
147  }
148  handle->i1 = handle->ui;
149 
150  /* control output */
151  handle->v1 = handle->up + handle->ui;
152  handle->Out = _IQsat(handle->v1, handle->Umax, handle->Umin);
153 }
