100 __STATIC_INLINE void PHASEVOLT_run(PHASEVOLT_Instance *handle)
101 {
102  /* Scale the incoming Modulation functions with the DC bus voltage value */
103  /* and calculate the 3 Phase voltages */
104  handle->temp = _IQmpy(handle->DcBusVolt, ONE_THIRD);
105  handle->VphaseA = _IQmpy(handle->temp,
106  (_IQmpy2(handle->MfuncV1) - handle->MfuncV2 - handle->MfuncV3));
107  handle->VphaseB = _IQmpy(handle->temp,
108  (_IQmpy2(handle->MfuncV2) - handle->MfuncV1 - handle->MfuncV3));
109 
110  /* Voltage transformation (a,b,c) -> (Alpha,Beta) */
111  handle->Valpha = handle->VphaseA;
112  handle->Vbeta = _IQmpy((handle->VphaseA
113  + _IQmpy2(handle->VphaseB)),INV_SQRT3);
114 }
