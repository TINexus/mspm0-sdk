130 __STATIC_INLINE void PWMGEN_run(PWMGEN_Instance *handle)
131 {
132  handle->dutyA = handle->HalfPerMax
133  - _IQmpy(handle->HalfPerMax, handle->MfuncC1);
134  handle->dutyB = handle->HalfPerMax
135  - _IQmpy(handle->HalfPerMax, handle->MfuncC2);
136  handle->dutyC = handle->HalfPerMax
137  - _IQmpy(handle->HalfPerMax, handle->MfuncC3);
138 
139  PWMGEN_checkLimits(&handle->dutyA, handle->dutyMax, handle->dutyMin);
140  PWMGEN_checkLimits(&handle->dutyB, handle->dutyMax, handle->dutyMin);
141  PWMGEN_checkLimits(&handle->dutyC, handle->dutyMax, handle->dutyMin);
142 }
