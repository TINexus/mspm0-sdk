374 __STATIC_INLINE DL_ADC12_IIDX HAL_processADCIRQ(ADC12_Regs *adcInst)
375 {
376  DL_ADC12_IIDX pendIrq = DL_ADC12_getPendingInterrupt(adcInst);
377  for(int x = 0;x < HAL_ADC_CHANNEL_MAX;x++)
378  {
379  if(adcInst == halAdc[x].inst && pendIrq == halAdc[x].endOfSeq)
380  {
381  halAdc[x].value = (uint16_t)HAL_getADCData(x);
382  }
383  }
384  return pendIrq;
385 }
