939 __STATIC_INLINE uint32_t DL_FlashCTL_getFlashSectorNumberInBank(
940  FLASHCTL_Regs *flashctl, uint32_t addr)
941 {
942  uint8_t numBanks = DL_FactoryRegion_getNumBanks();
943  uint32_t mainFlashSize = DL_FactoryRegion_getMAINFlashSize();
944  uint32_t sector, bankSectors, sectorInBank;
945 
946  /* Current sector over the whole memory map */
947  sector = DL_FlashCTL_getFlashSectorNumber(flashctl, addr);
948  if (numBanks > 1) {
949  /* Number of sectors per bank, considered max sector
950  * count for a bank-adjusted sector. Assume banks are
951  * evenly distributed */
952  bankSectors = (mainFlashSize / numBanks);
953  /* We will not assume that the maximum number of bank sectors is a
954  * multiple of 2. Which does mean incurring a modulo operation. */
955  sectorInBank = sector % bankSectors;
956  } else {
957  sectorInBank = sector;
958  }
959 
960  return sectorInBank;
961 }
