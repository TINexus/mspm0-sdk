105 __STATIC_INLINE uint32_t TRANSIT_run(TRANSIT_Instance *handle,
106  uint32_t finalVal, uint32_t refVal)
107 {
108  uint32_t out = 0;
109  int32_t tmp;
110  switch (handle->state)
111  {
112  case TRANSIT_STATE_COMPUTE_DIFF:
113  tmp = ((int32_t)finalVal - (int32_t)refVal)
114  >>TRANSIT_ANGLE_SCALE_FACTOR_TO_IQ10;
115  handle->diff = tmp;
116  handle->factor = TRANSIT_FACTOR_DELTA_INIT_VAL_IQ10;
117  handle->state = TRANSIT_STATE_IN_TRANSIT;
118  out = refVal;
119  break;
120  case TRANSIT_STATE_IN_TRANSIT:
121  out = finalVal - (uint32_t)(_IQ10mpy(handle->diff,
122  handle->factor)<<TRANSIT_ANGLE_SCALE_FACTOR_TO_IQ10);
123  handle->factor -= TRANSIT_FACTOR_DELTA_IQ10;
124  if(handle->factor < 0)
125  {
126  handle->state = TRANSIT_STATE_COMPLETE;
127  }
128  break;
129  case TRANSIT_STATE_COMPLETE:
130  out = finalVal;
131  break;
132  default:
133  /* This is expected to be empty */
134  break;
135  }
136  handle->out = out;
137  return out;
138 }
