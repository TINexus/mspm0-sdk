133 __STATIC_INLINE void PWMGEN_run(PWMGEN_Instance *handle)
134 {
135  handle->dutyA = handle->HalfPerMax
136  - _IQmpy(handle->HalfPerMax, handle->MfuncC1);
137  handle->dutyB = handle->HalfPerMax
138  - _IQmpy(handle->HalfPerMax, handle->MfuncC2);
139  handle->dutyC = handle->HalfPerMax
140  - _IQmpy(handle->HalfPerMax, handle->MfuncC3);
141 
142  PWMGEN_checkLimits(&handle->dutyA, handle->dutyMax, handle->dutyMin);
143  PWMGEN_checkLimits(&handle->dutyB, handle->dutyMax, handle->dutyMin);
144  PWMGEN_checkLimits(&handle->dutyC, handle->dutyMax, handle->dutyMin);
