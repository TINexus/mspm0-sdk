97 __STATIC_INLINE void RC_run(RMPCNTL_Instance *handle)
98 {
99  handle->Tmp = handle->TargetValue - handle->SetpointValue;
100  if (_IQabs(handle->Tmp) >= RMP_CNTRL_DELTA)
101  {
102  handle->RampDelayCount++;
103  if (handle->RampDelayCount >= handle->RampDelayMax)
104  {
105  if (handle->TargetValue >= handle->SetpointValue)
106  {
107  handle->SetpointValue += RMP_CNTRL_DELTA;
108  }
109  else
110  {
111  handle->SetpointValue -= RMP_CNTRL_DELTA;
112  }
113  handle->SetpointValue=_IQsat(handle->SetpointValue,
114  handle->RampHighLimit, handle->RampLowLimit);
115  handle->RampDelayCount = 0;
116  }
117  }
