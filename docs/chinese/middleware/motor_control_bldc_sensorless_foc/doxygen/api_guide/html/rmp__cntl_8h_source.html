121 __STATIC_INLINE void RC_run(RMPCNTL_Instance *handle)
122 {
123  handle->Tmp = handle->TargetValue - handle->SetpointValue;
124  if (_IQabs(handle->Tmp) >= handle->RampDelta)
125  {
126  handle->RampDelayCount++;
127  if (handle->RampDelayCount >= handle->RampDelayMax)
128  {
129  if (handle->TargetValue >= handle->SetpointValue)
130  {
131  handle->SetpointValue += handle->RampDelta;
132  }
133  else
134  {
135  handle->SetpointValue -= handle->RampDelta;
136  }
137  handle->SetpointValue =_IQsat(handle->SetpointValue,
138  handle->RampHighLimit, handle->RampLowLimit);
139  handle->RampDelayCount = 0;
140  }
141  }
142 }
